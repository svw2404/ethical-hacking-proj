# SECURE: authorization check before returning profile data
@app.route("/secure-profile")
def secure_profile():
    requested_id = request.args.get("id")
    current_user = session["user_id"]
    current_role = session["role"]

    # Only allow access to own profile, or admin accessing any
    if requested_id != str(current_user) and current_role != "admin":
        logger.warning(f"Blocked IDOR attempt: user={current_user} requested={requested_id}")
        return "403 Access Denied", 403

    # Safe to return the profile now
    profile = db.execute("SELECT * FROM profiles WHERE id = ?", (requested_id,)).fetchone()
    return render_template("secure_profile.html", profile=profile)
